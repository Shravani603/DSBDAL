#Import Required Libraries
import requests
import bs4
import pandas as pd

headers = {"User-Agent": "Mozilla/5.0"}


url = "https://www.flipkart.com/samsung-galaxy-s25-5g-icyblue-256-gb/p/itm3c1e495903acc"

response = requests.get(url, headers=headers)
print("Status Code:", response.status_code)

soup = bs4.BeautifulSoup(response.text, 'html.parser')

#title
title = soup.find('h1', {'class': 'v1zwn21l v1zwn26 _1psv1zeb9 _1psv1ze0'})
print("Product Name :", title.get_text(strip=True))

#price
price = soup.find('div', {'class': 'v1zwn21l v1zwn20 _1psv1zeb9 _1psv1ze0'})
print("Price :", price.get_text(strip=True))

#Product rating
ratings = soup.find_all('div', {'class': 'css-146c3p1'})
print("Overall Product Rating :", ratings[1].text)

#individual rating
individual_ratings = soup.find_all('div', class_='_1psv1zeb9 _1psv1ze0 _1psv1zee9 _1psv1zel0')

for rating in individual_ratings:
    print("Rating:",rating.text[0],"⭐")

#comment
comment_tags = soup.find_all('div', class_= '_1psv1zeb9 _1psv1ze0 _1psv1zee9 _1psv1zel0')

for comment in comment_tags:
    print("Comment:",comment.text[1:])

#customer names
filtered_customers = []

customer_name_tags = soup.find_all('div', class_= 'v1zwn21l v1zwn27 _1psv1zeb9 _1psv1ze0')

# Filter out non-customer-name strings like delivery info, services, etc.
unwanted_prefixes = [
    "10 Days", "Cash on", "Delivery", "Flipkart", "Assured",
    "Right Product", "No delivery", "Quick service", "by",
    "Free test", "Nearby", "All accessories", "Expert assistance",
    "Guaranteed help", "No hassle", "Quality checks","brand support",
    "6 step guidance", "Authorised service", "Expert help","128 GB + 12 GB"
]

for tag in customer_name_tags:
    text = tag.get_text(strip=True)

    if text == "Flipkart Customer" or not any(
        word in text for word in unwanted_prefixes
    ):
        filtered_customers.append(text)

for name in filtered_customers:
    print("Customer:", name)

#review
reviews = soup.find_all('div', class_='v1zwn21l v1zwn26 _1psv1zeb9 _1psv1ze0')

for review in reviews:
    print("Review:",review.get_text(strip=True))

#save to csv file
data = []

all_ratings = soup.find_all('div', class_='_1psv1zeb9 _1psv1ze0 _1psv1zee9 _1psv1zel0')
all_reviews = soup.find_all('div', class_='v1zwn21l v1zwn26 _1psv1zeb9 _1psv1ze0')
comment_tags = soup.find_all('div', class_='_1psv1zeb9 _1psv1ze0 _1psv1zee9 _1psv1zel0')

for customer, rating, comment, review in zip(
    filtered_customers,
    all_ratings,
    comment_tags,
    all_reviews
):

    data.append({
        "Customer Name": customer,
        "Rating": rating.text[0],
        "Comment Tag": comment.text[1:],
        "Review": review.get_text(strip=True)
    })

df = pd.DataFrame(data)

df.to_csv("samsung_s25_flipkart_reviews.csv", index=False)

print("Data saved successfully")
